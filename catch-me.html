# coding=utf-8

"""
    Catch Me Game
    A simple game where text escapes from the cursor
"""

import tkinter as tk
import random
import math


class CatchMeGame:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Catch Me Game")
        self.window.geometry("800x600")
        self.window.configure(bg="#1a1a1a")
        
        self.width = 800
        self.height = 600
        self.escape_distance = 100
        self.attempts = 0
        self.escapes = 0
        
        # Create canvas
        self.canvas = tk.Canvas(
            self.window,
            width=self.width,
            height=self.height,
            bg="#1a1a1a",
            highlightthickness=0
        )
        self.canvas.pack()
        
        # Initial text position
        self.text_x = self.width // 2
        self.text_y = self.height // 2
        
        # Create text
        self.text_id = self.canvas.create_text(
            self.text_x,
            self.text_y,
            text="Catch me!",
            font=("Arial", 30, "bold"),
            fill="#4ade80"
        )
        
        # Create stats labels
        self.stats_frame = tk.Frame(self.window, bg="#1a1a1a")
        self.stats_frame.pack(pady=10)
        
        self.attempts_label = tk.Label(
            self.stats_frame,
            text="Attempts: 0",
            font=("Arial", 12),
            bg="#1a1a1a",
            fg="#fff"
        )
        self.attempts_label.pack(side=tk.LEFT, padx=20)
        
        self.escapes_label = tk.Label(
            self.stats_frame,
            text="Escapes: 0",
            font=("Arial", 12),
            bg="#1a1a1a",
            fg="#fff"
        )
        self.escapes_label.pack(side=tk.LEFT, padx=20)
        
        # Bind mouse movement
        self.canvas.bind("", self.on_mouse_move)
        
    def get_distance(self, x1, y1, x2, y2):
        """Calculate distance between two points"""
        return math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)
    
    def escape_to_new_position(self):
        """Move text to a random new position"""
        margin = 50
        self.text_x = random.randint(margin, self.width - margin)
        self.text_y = random.randint(margin, self.height - margin)
        self.escapes += 1
        self.escapes_label.config(text="Escapes: {}".format(self.escapes))
        
        # Update text position
        self.canvas.coords(self.text_id, self.text_x, self.text_y)
    
    def on_mouse_move(self, event):
        """Handle mouse movement"""
        self.attempts += 1
        self.attempts_label.config(text="Attempts: {}".format(self.attempts))
        
        cursor_x = event.x
        cursor_y = event.y
        
        distance = self.get_distance(
            cursor_x, cursor_y,
            self.text_x, self.text_y
        )
        
        if distance < self.escape_distance:
            self.escape_to_new_position()
    
    def run(self):
        """Start the game"""
        self.window.mainloop()


if __name__ == '__main__':
    game = CatchMeGame()
    game.run()
